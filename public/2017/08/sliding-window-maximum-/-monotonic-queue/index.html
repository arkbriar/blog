Given an array nums, there is a sliding window of size k which is moving from the very left of the array to the very right. You can only see the k numbers in the window. Each time the sliding window moves right by one position.

For example, Given nums = [1,3,-1,-3,5,3,6,7], and k = 3.

Therefore, return the max sliding window as [3,3,5,5,6,7].

 1 vector<int> maxSlidingWindow(vector<int> &nums, int k) {
 2     vector<int> res;
 3     // tracking index instead of value
 4     deque<int> dq;
 5     
 6     for (int i = 0; i < nums.size(); ++ i) {
 7         // dequeue (i - k)th element if exists
 8         if (!dq.empty() && dq.front() == i - k) dq.pop_front();
 9         // remove those less than current
10         while (!dq.empty() && nums[dq.back()] < nums[i]) dq.pop_back();
11         // enqueue current
12         dq.push_back(i);
13         
14         if (i >= k - 1) res.push_back(nums[dq.front()]);
15     }
16     
17     return res;
18 }
